Window窗口

介绍

window窗口

示例代码

Ext.define('App.demo.window.View',{
	
	init: function(){
		var view = Ext.create('Ins.window.Window',{
			title:"我的Window",
			views : [
				App.getView('App.demo.window.Form')
			],
			//标题工具按钮
			headButtons : [{
			    type:'help',
			    tooltip: '获取帮助文档',
			    callback: function(panel, tool, event) {
			    	Message.msg("帮助文档");
			    }
			}],
			//工具栏按钮
			actionButtons : [{
				text: '保存',
					iconCls: 'icon-save',
					handler: function(){
						Message.msg("保存");
					}
			},{
				text: '对其到中央方法',
					iconCls: 'icon-save',
					handler: function(){
						//对齐到中央
						view.alignToCenter();
					}
			}],
			//工具栏按钮方向
			actionButtonsAlign : 'bottom',
			//工具栏样式 footer/default 。。。。
			actionButtonsStyle: 'footer',
			//窗口显示前事件
			onBeforeShow : function(me, eOpts){
				alert("触发事件:windown的onBeforeShow");
				console.log("触发事件:windown的onBeforeShow");
				console.log(me);
				console.log(eOpts);
				//返回true表示可以打开，false将阻止窗口打开。void和true等效；
				return true;
			},
			//窗口关闭前事件
			onBeforeHide : function(me, eOpts){
				alert("触发事件:windown的onBeforeHide，返回false将阻止窗口影藏");
				console.log("触发事件:windown的onBeforeHide");
				console.log(me);
				console.log(eOpts);
				//返回true表示可以隐藏，false将阻止窗口隐藏。void和true等效；
				return false;
			}
		});
		view.show();
		return null;
	}	
});

属性

属性名属性值类型描述默认值
title String 窗口标题 未命名窗体
views view[ ] 面板组件 [ ]
hideTitle boolean 是否隐藏标题 false
headButtons button[] 标题工具按钮 []
actionButtons button[] 工具栏按钮 []
actionButtonsAlign String 工具栏按钮方向 'bottom'
actionButtonsStyle String 工具栏样式 footer/default 。。。。 'footer'
useAppBg boolean 使用背景色 false
closeAction String 点击关闭工具时的操作 'hide'
collapsible boolean 是否显示折叠工具 true
resizable boolean 是否允许缩放 true
modal boolean 是否显示遮罩 false
layout String 布局 'fit'
headerPosition String 标题方向 'top'
maximizable boolean 是否显示最大化工具 false
minimizable boolean 是否显示最小化工具 false
constrain boolean 添加约束 true
constrainHeader boolean 和constrain配合使用 true

方法

返回值方法名称描述
void alignToCenter() 将窗口移动至当前浏览器中间显示
void show() 显示窗口
void hide() 隐藏窗口

事件

返回值事件名称绑定方法描述
boolean onBeforeShow function(me, eOpts){ } 窗口显示前触发的事件，参数类型me组件本身，返回true表示允许窗口打开，false将阻止窗口打开。void和true等效；
boolean onBeforeHide function(me, eOpts){ } 窗口关闭前触发的事件，参数类型me组件本身，返回true表示允许窗口隐藏，false将阻止窗口隐藏。void和true等效；
