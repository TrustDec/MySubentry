AccrodionLayout(布局)

介绍

手风琴布局

示例代码

Ext.define('App.demo.layout.accrodionlayout.View',{
	
	init: function(){
		
		var view = Ext.create('Ins.layout.AccrodionLayout',{
			title:'AccrodionLayout布局',
			//存放区域视图
			views: [
				App.getView('App.demo.layout.accrodionlayout.Panel1'),
				App.getView('App.demo.layout.accrodionlayout.Panel2'),
				App.getView('App.demo.layout.accrodionlayout.Panel3'),
				App.getView('App.demo.layout.accrodionlayout.Panel4')
			]
		});
		return view;
	}
});
