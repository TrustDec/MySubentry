HiddenField

介绍

隐藏域

示例代码

Ext.define('App.demo.form.field.HiddenField',{
	
	init: function(){
		var me = this;
		var view = Ext.create('Ins.form.Form',{
			fields : [{
					text:"隐藏域",
					name:"name",
					xtype:"ins_hiddenfield",
					value:'地瓜地瓜',
					//改变内容事件  me:Ext.Component;改变后的值newValue，改变之前的值oldValue
					onFieldChange : function(me, newValue, oldValue){
						console.log(me);
						alert("内容改变事件    新值:"+newValue+" 旧值:"+oldValue);
					}
			}],
			actionButtons : [{
				text:'给隐藏域赋值',
				handler: function(){
					Message.input("请输入",function(flag, val){
						if(flag){
							view.setValue("name",val);
						}
					});
				}
			}]
		});
		return view;
	}
});
