BorderLayout(布局)

介绍

该布局有5个区域:上、下、左、右、中五个区域。

示例代码

Ext.define('App.demo.layout.borderlayout.View',{
	
	init: function(){
		var right = App.getView('App.demo.layout.borderlayout.Right');
		var view = Ext.create('Ins.layout.BorderLayout',{
			//标题
			title:'BorderLayout布局',
			//定义组件位置
			location: ['top','bottom','left','right','center'],
			//存放区域视图
			views: [
				App.getView('App.demo.layout.borderlayout.Top'),
				App.getView('App.demo.layout.borderlayout.Bottom'),
				App.getView('App.demo.layout.borderlayout.Left'),
				right,
				App.getView('App.demo.layout.borderlayout.Center')
			],
			//定义组件是否可折叠
			fold: [false,false,true,true,false]
		});
		//设置左边区域折叠
		right.collapse();
		return view;
	}
	
});

类名

Ins.layout.BorderLayout

别名

alias : ["widget.ins_borderlayout"]

继承自

Ext.panel.Panel

构造对象

var view = Ext.create('Ins.layout.BorderLayout',{
			//标题
			title:'',
			//定义组件位置
			location: [],
			//存放区域视图
			views: [],
			//定义组件是否可折叠
			fold: []
		});

属性

属性名属性值类型描述默认值
title String 标题 默认为空不显示
location String[] 上下左右中五个区域,至少选一个区域['top','bottom','left','right','center'] 无
views view[] 存放对应区域显示的view视图 无
fold boolean[] 设置对应区域是否可折叠true：折叠，false：不折叠 全为false
useAppBg boolean 是否使用系统指定背景 false
border boolean 是否显示边框 false
